// INSERTION_SORT(CRESCENTE) //

INÍCIO

	DEFINIR FUNÇÃO TROCA QUE RECEBE, COMO PARÂMETROS, LISTA, INDICE1, INDICE2.

	DEFINIR VARIÁVEL TEMPORARIO = LISTA[INDICE1]

	LISTA[INDICE1]	= LISTA[INDICE2]

	LISTA[INDICE2] = TEMPORARIO

	FIM FUNÇÃO TROCA


	DEFINIR A FUNÇÃO INSERTION_SORT QUE RECEBE COMO PARÂMETRO, LISTA.

	DEFINIR VARÍAVEL x = 1

	PARA x EM ALCANCE DE TAMANHO DA LISTA 

		DEFINIR A VARIÁVEL VALOR = LISTA[x]

		DEFINIR VARIÁVEL i = ÍNDICE DE VALOR - 1

		ENQUANTO i >= 0
	
			SE LISTA[i] > VALOR

				TROCA(LISTA, ÍNDICE DE VALOR, INDICE DE LISTA[i])
		
			i -= 1


			FIM SE

	FIM FUNÇÃO INSERTION_SORT
	

FIM
